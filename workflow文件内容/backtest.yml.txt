name: 策略回测
on:
  workflow_dispatch: {}       # 只手动触发
permissions:
  contents: write
jobs:
  backtest:
    runs-on: ubuntu-latest
    timeout-minutes: 120
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
        with:
          python-version: '3.12'
      - run: pip install -r requirements.txt
      - run: python -m src.main backtest
      - name: 提交结果
        run: |
          git config user.name "scanner-bot"
          git config user.email "bot@users.noreply.github.com"
          git add docs
          git diff --cached --quiet || git commit -m "回测 $(date +%F)"
          git push
